function uniquementLettres(x)
{
	// on peut utiliser cette fonction pour vérifier si le paramètre x est composé uniquement par des lettres alphabétiques
	x=x.toUpperCase()
    var i=0
    while (x.charAt(i)>="A" && x.charAt(i)<="Z" && i<x.length) 
				i++
	return i == x.length
}
